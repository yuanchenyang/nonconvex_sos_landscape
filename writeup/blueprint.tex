\section{Statement and setup}\label{sec:setup}
We work in dehomogenized coordinates: a ternary quartic is viewed as a quartic
polynomial in two affine variables \(x_0,x_1\). The optimization variable is a
rank-$4$ Burer--Monteiro factor
\[
  u=(u_0,u_1,u_2,u_3)\in \Quad^4.
\]
All tangent directions are subject to the same quadratic degree bound.

\begin{definition}[Basic maps; Lean: \texttt{sigma}, \texttt{A}, \texttt{residual}, \texttt{IsSOCP}]
\label{def:basic-maps}
For \(u,v\in \Quad^4\) and \(p\in \Quart\), define
\[
  \sig(u)=\sum_{i=0}^3 u_i^2,\qquad
  A_u(v)=\sum_{i=0}^3 u_i v_i,\qquad
  r(u,p)=\sig(u)-p.
\]
Fix a positive-definite bilinear form \(B\) on \(\Poly\). We say that \(u\) is
a first-order critical point of \(p\) if
\[
  B\bigl(A_u(v),\,r(u,p)\bigr)=0
  \qquad\text{for all }v\in \Quad^4,
\]
and a second-order critical point if, in addition,
\[
  B\bigl(\sig(v),\,r(u,p)\bigr)+2B\bigl(A_u(v),A_u(v)\bigr)\ge 0
  \qquad\text{for all }v\in \Quad^4.
\]
\end{definition}

\begin{definition}[SOS quartic target; Lean: \texttt{IsSOSQuartic}]
\label{def:sos}
An \emph{SOS quartic} is a polynomial \(p\in \Quart\) admitting a finite
representation
\[
  p=\sum_{j=1}^k q_j^2,\qquad q_j\in \Quad .
\]
\end{definition}

\begin{definition}[Relation map and exact affine relations; Lean: \texttt{relationPoly},
\texttt{relationPolyLin}, \texttt{exactAffineSubmodule}]
\label{def:relation-poly-lin}
For fixed \(u\in \Quad^4\), define the relation map
\[
  \rel_u:\R^4\to \Poly,\qquad \rel_u(c)=\sum_{i=0}^3 c_i u_i.
\]
This is the linear map called \texttt{relationPolyLin} in Lean. Its kernel is
the space of scalar linear dependences among the four coordinates of \(u\).

Define the \emph{exact affine relation space}
\[
  \Exact(u)=\{c\in \R^4:\rel_u(c)\in \Aff\}.
\]
Equivalently, \(\Exact(u)\) is the kernel of the map taking
\(\rel_u(c)\) to its homogeneous quadratic coefficients.
\end{definition}

\begin{remark}\label{rem:goal}
The theorem proved in Lean is the statement that every admissible rank-$4$
\(\SOCP\) over an SOS quartic has zero residual:
\[
  r(u,p)=0.
\]
Equivalently, there are no spurious second-order critical points in the
rank-$4$ ternary quartic problem.
\end{remark}

\section{The certificate mechanism}\label{sec:certificate}
The proof follows the certificate philosophy already used in the univariate
writeup: one shows that every SOS square \(q^2\) can be written as an image term
for \(A_u\) plus kernel-square corrections. The \(\SOCP\) inequalities then
force the residual to vanish.

\begin{lemma}[Abstract image-plus-kernel-square certificate; Lean:
\texttt{admissible\_image\_plus\_cone\_residual\_eq\_zero}]
\label{lem:certificate}
Let \(u\in \Quad^4\) be a \(\SOCP\) for an SOS quartic \(p\). Assume that
\[
  p=q_{\mathrm{img}}+\sum_{m=1}^N \sig(w^{(m)}),
\]
where \(q_{\mathrm{img}}\in \im(A_{u_{\mathrm{img}}})\) for some admissible
\(u_{\mathrm{img}}\), each \(w^{(m)}\in \Quad^4\) satisfies
\(A_u(w^{(m)})=0\), and \(r(u,p)\) is orthogonal to
\(\im(A_{u_{\mathrm{img}}})\). Then \(r(u,p)=0\).
\end{lemma}

\begin{proof}
Set \(r=r(u,p)\). By first-order criticality, the image term pairs to zero:
\[
  B(q_{\mathrm{img}},r)=0.
\]
For each kernel direction \(w=w^{(m)}\), the Hessian inequality gives
\[
  B(\sig(w),r)+2B(A_u(w),A_u(w))\ge 0.
\]
Since \(A_u(w)=0\), this reduces to \(B(\sig(w),r)\ge 0\). Summing over
\(m\) yields
\[
  B(p,r)\ge 0.
\]
On the other hand, first-order criticality with \(v=u\) gives
\[
  B(\sig(u),r)=0.
\]
Therefore
\[
  B(r,r)=B(\sig(u)-p,r)=-B(p,r)\le 0.
\]
Positive-definiteness of \(B\) implies \(B(r,r)\ge 0\), hence \(B(r,r)=0\),
and therefore \(r=0\).
\end{proof}

\begin{corollary}[Image case]\label{cor:image}
If \(p\in \im(A_u)\), then \(r(u,p)=0\).
\end{corollary}

\begin{proof}
Apply \cref{lem:certificate} with no kernel-square terms and
\(u_{\mathrm{img}}=u\).
\end{proof}

\begin{corollary}[Constant scalar relation; Lean:
\texttt{residual\_eq\_zero\_of\_constant\_relation}]
\label{cor:constant-relation}
If there exists \(0\neq c\in \R^4\) such that \(\rel_u(c)=0\), then
\(r(u,p)=0\) for every SOS quartic \(p\).
\end{corollary}

\begin{proof}
Write \(p=\sum_j q_j^2\) as in \cref{def:sos}. For each \(j\), define
\[
  w^{(j)}=(c_0 q_j,\dots,c_3 q_j)\in \Quad^4.
\]
Then \(A_u(w^{(j)})=q_j\rel_u(c)=0\), while
\[
  \sig(w^{(j)})=\Bigl(\sum_{i=0}^3 c_i^2\Bigr)q_j^2.
\]
Since \(\sum_i c_i^2>0\), each \(q_j^2\) is a positive scalar multiple of a
kernel square. Summing over \(j\) produces the decomposition required by
\cref{lem:certificate}.
\end{proof}

\begin{proposition}[Equivariance under affine changes and orthogonal mixing;
Lean families in \texttt{AffineSocpTransform.lean} and
\texttt{FactorTransform.lean}]
\label{prop:equivariance}
The properties of being admissible, being an SOS quartic, being a \(\SOCP\),
and having vanishing residual are preserved by
\begin{enumerate}
\item invertible affine changes of variables in \(x_0,x_1\), and
\item orthogonal changes of basis in the four-dimensional coefficient space of
the factor coordinates.
\end{enumerate}
\end{proposition}

\begin{proof}
Both operations preserve the quadratic degree bound, commute with
\(\sig\), transport \(A_u\) to the corresponding linearized map in the new
coordinates, and transport the bilinear form by pullback. Hence they preserve
the first- and second-order criticality conditions and the equality
\(r(u,p)=0\).
\end{proof}

\section{Three global closure mechanisms}\label{sec:global-closures}
The full proof uses three conceptual closure results.

\begin{theorem}[Span-three certificate; Lean:
\texttt{residual\_eq\_zero\_of\_contains\_aff1}]
\label{thm:contains-aff1}
Assume there exist \(c^{(0)},c^{(1)},c^{(2)}\in \R^4\) such that
\[
  \rel_u(c^{(0)})=1,\qquad
  \rel_u(c^{(1)})=x_0,\qquad
  \rel_u(c^{(2)})=x_1.
\]
Then \(r(u,p)=0\) for every SOS quartic \(p\).
\end{theorem}

\begin{proof}
Fix a quadratic \(q\in \Quad\). Write
\[
  q=q_{\le 1}+q_{\mathrm{hom}},\qquad
  q_{\mathrm{hom}}=x_0 a+x_1 b
\]
with \(a,b\in \Aff\). Define the raw kernel direction
\[
  w_{\mathrm{raw}}
  =\rel(-c^{(0)},q_{\mathrm{hom}})
   +\rel(c^{(1)},a)
   +\rel(c^{(2)},b),
\]
where \(\rel(c,h)\) means the admissible direction with \(i\)-th coordinate
\(c_i h\). By construction,
\[
  A_u(w_{\mathrm{raw}})
  =-q_{\mathrm{hom}}+x_0 a+x_1 b
  =0.
\]
Now expand \(\sig(w_{\mathrm{raw}})\). The quartic contribution comes entirely
from the square of the first summand:
\[
  \sig(w_{\mathrm{raw}})_{\deg 4}
  =\Bigl(\sum_{i=0}^3 (c_i^{(0)})^2\Bigr) q_{\mathrm{hom}}^2.
\]
Every other term contains at least one factor \(a\) or \(b\), hence has degree
at most \(3\). After scaling \(w_{\mathrm{raw}}\) by
\[
  \alpha=\Bigl(\sum_i (c_i^{(0)})^2\Bigr)^{-1/2},
\]
we obtain a kernel direction \(w\) such that
\[
  q^2-\sig(w)\in \Poly_{\le 3}.
\]
Since \(1,x_0,x_1\in W(u):=\Span_\R\{u_0,u_1,u_2,u_3\}\), every cubic lies in
\(\im(A_u)\): multiply the exact relations for \(1,x_0,x_1\) by suitable
linear polynomials. Thus \(q^2\) is an image term plus one kernel square.

Applying this construction to each quadratic summand in an SOS decomposition of
\(p\) and summing gives the decomposition in \cref{lem:certificate}. Hence
\(r(u,p)=0\).
\end{proof}

\begin{theorem}[Mixed-affine coarse theorem; Lean:
\texttt{residual\_eq\_zero\_of\_relations\_const\_affineLine}]
\label{thm:const-affine-line}
Assume there exist \(c^{(0)},c^{(1)}\in \R^4\) such that
\[
  \rel_u(c^{(0)})=1,\qquad
  \rel_u(c^{(1)})=r+a x_0+b x_1,
\]
with \(a^2+b^2\neq 0\). Then \(r(u,p)=0\) for every SOS quartic \(p\).
\end{theorem}

\begin{proof}
By \cref{prop:equivariance}, we may apply an affine change of variables sending
\(r+a x_0+b x_1\) to \(x_0\). Thus it is enough to treat the normalized case
where \(1\) and \(x_0\) are exact relations.

The remaining two coordinates then define a quadratic plane in \(\HomQ\). There
are only three possibilities.
\begin{enumerate}
\item The plane is surjective: an invertible basis of either
\(\Span\{x_0^2,x_1^2\}\) or \(\Span\{x_0x_1,x_0^2-x_1^2\}\). Then every quartic
lies in \(\im(A_u)\), so \cref{cor:image} applies.
\item The plane is of rank \(14\), equivalently an invertible basis of
\(\Span\{x_0x_1,x_1^2\}\). Then the only missing quotient class is \(x_0^4\),
and it is supplied by the explicit kernel square
\[
  \sig\!\bigl(\rel(-c_2,t x_0x_1)+\rel(c_3,t x_0^2)\bigr)
\]
for the two plane relations \(c_2,c_3\).
\item The plane is of rank \(13\), equivalently an invertible basis of
\(\Span\{x_0^2,x_0x_1\}\). Then the only missing quotient classes are
\(x_1^3\) and \(x_1^4\), and both are supplied by the kernel family
\[
  \rel(-c_2,x_1\ell)+\rel(c_3,x_0\ell),
  \qquad \ell=b+c x_1 .
\]
\end{enumerate}
In every case, each SOS square is an image term plus kernel-square corrections,
so \cref{lem:certificate} yields \(r(u,p)=0\). Transport back by
\cref{prop:equivariance}.
\end{proof}

\begin{theorem}[Surjective homogeneous-basis theorem; Lean:
\texttt{exact\_relations\_of\_homQuadBasis\_det} and
\texttt{residual\_eq\_zero\_of\_relations\_const\_homQuadBasis\_det}]
\label{thm:surjective-basis}
Assume there exists \(c^{(0)}\in \R^4\) with \(\rel_u(c^{(0)})=1\), and three
further relations \(d^{(0)},d^{(1)},d^{(2)}\in \R^4\) whose homogeneous
quadratic parts form an invertible \(3\times 3\) matrix in the basis
\((x_0^2,x_0x_1,x_1^2)\). Then \(r(u,p)=0\) for every SOS quartic \(p\).
\end{theorem}

\begin{proof}
Invertibility of the homogeneous coefficient matrix lets us solve back for
exact relations with homogeneous parts exactly \(x_0^2\), \(x_0x_1\), and
\(x_1^2\). Since \(1\) is already an exact relation, every quartic monomial
lies in \(\im(A_u)\):
\begin{itemize}
\item degree \(\le 2\) monomials come from multiplying the relation \(1\) by a
quadratic,
\item cubic monomials come from multiplying one of \(x_0^2,x_0x_1,x_1^2\) by a
linear polynomial,
\item quartic monomials come from multiplying one of \(x_0^2,x_0x_1,x_1^2\) by
a quadratic monomial.
\end{itemize}
Thus every quartic \(p\) belongs to \(\im(A_u)\), and \cref{cor:image} applies.
\end{proof}

\begin{theorem}[Low-affine plane theorem; Lean families in
\texttt{RepresentativeLowAffine.lean}]
\label{thm:low-affine}
Assume there are exact relations for \(x_0\) and \(x_1\), together with two
more quadratic relations whose homogeneous parts span a nonzero plane in
\(\HomQ\). Then every SOS quartic \(p\) satisfies \(r(u,p)=0\).
\end{theorem}

\begin{proof}
The homogeneous plane is classified by the three $2\times 2$ minors of its
coefficient matrix. If the cross determinant vanishes, the plane has a common
factor and reduces to the common-factor charts. If the completed-square
discriminant vanishes, it reduces to the cross chart. Otherwise it reduces,
after shear and scaling, to either the diagonal-sum or diagonal-difference
chart. In each chart the image is explicitly described and the missing quotient
classes are supplied by equally explicit kernel-square families. This is
exactly the low-affine analogue of the rank-$13$ and rank-$14$ mixed-affine
arguments. Applying \cref{lem:certificate} in each normal form proves the
theorem.
\end{proof}

\begin{theorem}[Affine-rank-one closure after line normalization; Lean families
\texttt{residual\_eq\_zero\_of\_exactAffineDimOne\_affineLine\_tailRangeZero},
\texttt{...tailRangeOne}, \texttt{...tailRangeTwo}]
\label{thm:affine-rank-one}
Assume \(\dim \Exact(u)=1\), that \(\Exact(u)\) contains a nonconstant affine
line but no constant relation, and normalize that line to \(x_0\). Let
\(\Tail(u)\) be the space of exact affine relations with zero \(x_0\)
coefficient, projected to their \((1,x_1)\)-tails. Then
\(\dim \Tail(u)\in \{0,1,2\}\), and in each of the three cases every SOS quartic
satisfies \(r(u,p)=0\).
\end{theorem}

\begin{proof}
If \(\dim \Tail(u)=0\), all complementary relations are homogeneous, giving the
linear low-affine endpoint.

If \(\dim \Tail(u)=1\), there is a common one-dimensional affine tail. After
normalization one reaches a finite list of affine-rank-one charts: repeated
line, shared-tail, common-factor, and diagonal endpoints. Each endpoint has an
explicit quartic image theorem and an explicit kernel-square correction for the
missing quotient classes.

If \(\dim \Tail(u)=2\), the tails span \(\Span\{1,x_1\}\). The corresponding
tail matrix either has nonzero determinant, which is the generic range-two
case, or determinant zero. In the determinant-zero regime, a shear reduction
produces either the shared-tail family or the \(x_0^2\)-tail family; a
discriminant then splits each of these into the zero and nonzero subcases.
Every one of these finitely many endpoints is solved by an explicit
image-plus-kernel-square certificate. Hence \(r(u,p)=0\) in all three tail
ranks.
\end{proof}

\section{The exact-affine classifier}\label{sec:classifier}
We now explain the classifier that handles the linearly independent case.

\begin{lemma}[Dimension bounds for exact affine relations]
\label{lem:dim-bounds}
Assume \(\Ker(\rel_u)=\{0\}\). Then
\[
  1\le \dim \Exact(u)\le 3.
\]
Hence \(\dim \Exact(u)\in \{1,2,3\}\).
\end{lemma}

\begin{proof}
By \cref{def:relation-poly-lin}, \(\Exact(u)\) is the kernel of the homogeneous
coefficient map
\[
  \R^4\longrightarrow \HomQ\simeq \R^3.
\]
Since the domain has dimension \(4\) and the codomain has dimension \(3\),
rank--nullity gives \(\dim\Exact(u)\ge 1\). On the other hand, because
\(\Ker(\rel_u)=\{0\}\), the affine coefficient map on \(\Exact(u)\) is
injective, so \(\dim\Exact(u)\le \dim \Aff =3\).
\end{proof}

\begin{theorem}[Exact-affine dimension \(3\); Lean:
\texttt{residual\_eq\_zero\_of\_exactAffineDimThree}]
\label{thm:exact-dim3}
If \(\Ker(\rel_u)=\{0\}\) and \(\dim \Exact(u)=3\), then \(r(u,p)=0\).
\end{theorem}

\begin{proof}
The affine coefficient map \(\Exact(u)\to \Aff\) is injective and both domain
and codomain have dimension \(3\), so it is an isomorphism. Therefore there are
exact relations for \(1,x_0,x_1\). Apply \cref{thm:contains-aff1}.
\end{proof}

\begin{theorem}[Exact-affine dimension \(2\) with a constant relation; Lean:
\texttt{residual\_eq\_zero\_of\_exactAffineDimTwo\_contains\_one}]
\label{thm:exact-dim2-const}
If \(\Ker(\rel_u)=\{0\}\), \(\dim \Exact(u)=2\), and \(1\in \rel_u(\Exact(u))\),
then \(r(u,p)=0\).
\end{theorem}

\begin{proof}
Fix an exact constant relation \(1\). Since \(\Exact(u)\) has dimension \(2\),
there is a second independent exact affine relation \(r+a x_0+b x_1\) with
\(a^2+b^2\neq 0\). Apply \cref{thm:const-affine-line}.
\end{proof}

\begin{theorem}[Exact-affine dimension \(2\) without a constant relation; Lean:
\texttt{residual\_eq\_zero\_of\_exactAffineDimTwo\_noConst}]
\label{thm:exact-dim2-noconst}
If \(\Ker(\rel_u)=\{0\}\), \(\dim \Exact(u)=2\), and
\(1\notin \rel_u(\Exact(u))\), then \(r(u,p)=0\).
\end{theorem}

\begin{proof}
First extract an affine pair \(x_0+r_0\), \(x_1+r_1\) from \(\Exact(u)\), then
translate by \((x_0,x_1)\mapsto (x_0-r_0,x_1-r_1)\). After this translation,
we have exact relations for \(x_0\) and \(x_1\), and every remaining exact
affine relation has zero linear part.

Consider the translated zero-linear-tail kernel. There are two possibilities.
\begin{enumerate}
\item Every such relation has zero constant term. Then the complementary
quadratic relations form a genuine homogeneous plane in \(\HomQ\), so
\cref{thm:low-affine} applies.
\item Some translated kernel relation has nonzero constant term. Then choose a
pair of translated relations with constants \(1\) and \(0\). If their
homogeneous parts are dependent, a linear combination produces a constant
relation and one falls back to \cref{thm:contains-aff1}. If their homogeneous
parts are independent, the constant-perturbed low-affine plane is classified by
the same common-factor/cross/diagonal normal forms as in
\cref{thm:low-affine}, now with one relation shifted by \(1\). The formalized
constant-perturbed low-affine plane theorems close these cases.
\end{enumerate}
Thus \(r(u,p)=0\).
\end{proof}

\begin{theorem}[Exact-affine dimension \(1\) with a constant relation; Lean:
\texttt{residual\_eq\_zero\_of\_exactAffineDimOne\_contains\_one}]
\label{thm:exact-dim1-const}
If \(\dim \Exact(u)=1\) and \(1\in \rel_u(\Exact(u))\), then \(r(u,p)=0\).
\end{theorem}

\begin{proof}
Let \(c^{(0)}\) generate \(\Exact(u)\) with \(\rel_u(c^{(0)})=1\). Consider
the subspace
\[
  K=\{c\in \R^4 : [1]\rel_u(c)=0\}.
\]
Since the constant coefficient map is surjective, \(\dim K=3\). The projection
from \(K\) to homogeneous quadratic coefficients is injective: if a vector in
\(K\) also has zero homogeneous quadratic part, then it would give a second
independent exact affine relation, contradicting \(\dim\Exact(u)=1\). Thus the
three vectors in \(K\) provide three independent quadratic relations.

Now inspect their affine tails. The map sending a relation in \(K\) to its
\((x_0,x_1)\)-tail has rank \(0\), \(1\), or \(2\). In each case, elementary
row operations inside \(K\) produce a basis whose relations are of the form
\[
  \text{(affine tail)}+x_0^2,\qquad
  \text{(affine tail)}+x_0x_1,\qquad
  \text{(affine tail)}+x_1^2.
\]
Their homogeneous coefficient matrix is therefore invertible. Applying
\cref{thm:surjective-basis} yields \(r(u,p)=0\).
\end{proof}

\begin{theorem}[Exact-affine dimension \(1\) without a constant relation; Lean:
\texttt{residual\_eq\_zero\_of\_exactAffineDimOne\_noConst}]
\label{thm:exact-dim1-noconst}
If \(\Ker(\rel_u)=\{0\}\), \(\dim \Exact(u)=1\), and
\(1\notin \rel_u(\Exact(u))\), then \(r(u,p)=0\).
\end{theorem}

\begin{proof}
Extract the unique exact affine line and normalize it by an affine equivalence
to \(x_0\). After this normalization, the remaining exact affine relations with
zero \(x_0\)-coefficient define the tail space \(\Tail(u)\subseteq\Span\{1,x_1\}\).
Its dimension is \(0\), \(1\), or \(2\). The three cases are exactly the three
tail-rank branches in \cref{thm:affine-rank-one}, so \(r(u,p)=0\).
\end{proof}

\begin{theorem}[Exact-affine classifier; Lean:
\texttt{residual\_eq\_zero\_of\_relationPolyLin\_ker\_eq\_bot}]
\label{thm:classifier}
If \(\Ker(\rel_u)=\{0\}\), then \(r(u,p)=0\) for every SOS quartic \(p\) and
every \(\SOCP\) \(u\).
\end{theorem}

\begin{proof}
By \cref{lem:dim-bounds}, \(\dim \Exact(u)\in\{1,2,3\}\).
\begin{itemize}
\item If \(\dim \Exact(u)=3\), apply \cref{thm:exact-dim3}.
\item If \(\dim \Exact(u)=2\), split according to whether \(1\in \rel_u(\Exact(u))\)
and apply \cref{thm:exact-dim2-const,thm:exact-dim2-noconst}.
\item If \(\dim \Exact(u)=1\), split according to whether \(1\in \rel_u(\Exact(u))\)
and apply \cref{thm:exact-dim1-const,thm:exact-dim1-noconst}.
\end{itemize}
These cases exhaust the possibilities.
\end{proof}

\section{Proof of the main theorem}\label{sec:main-proof}
\begin{theorem}[No spurious rank-$4$ SOCPs for ternary quartics; Lean:
\texttt{ternaryQuartic\_rankFour\_no\_spurious\_socp}]
\label{thm:main}
Let \(p\) be an SOS quartic, let \(u\in \Quad^4\), and let \(B\) be a
positive-definite bilinear form on \(\Poly\). If \(u\) is a second-order
critical point of \(p\), then
\[
  r(u,p)=0.
\]
\end{theorem}

\begin{proof}
Split into two cases according to \(\Ker(\rel_u)\).

\smallskip
\noindent\textbf{Case 1: \(\Ker(\rel_u)\neq \{0\}\).}
Then there exists \(0\neq c\in \R^4\) with \(\rel_u(c)=0\). By
\cref{cor:constant-relation}, \(r(u,p)=0\).

\smallskip
\noindent\textbf{Case 2: \(\Ker(\rel_u)=\{0\}\).}
Then \cref{thm:classifier} applies directly and gives \(r(u,p)=0\).

\smallskip
This proves the theorem.
\end{proof}

\begin{remark}\label{rem:structure}
The proof has the same overall shape as the univariate argument but with a
different algebraic middle layer. The univariate proof closes by a single
divisibility argument on the common factor \(g\). In the ternary quartic case,
the multivariate replacement is the exact-affine classifier:
\[
  \text{dependent relation}
  \;\;\text{or}\;\;
  \dim \Exact(u)\in \{1,2,3\},
\]
followed by the span-three, mixed-affine, low-affine, and affine-rank-one
closure theorems described above.
\end{remark}
